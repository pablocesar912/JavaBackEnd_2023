		"a":

				Leer n

		"b":
			si c=0 Entonces
				Escribir "Ingrese el tamaño del vector"

					vb(i)=Aleatorio(-100,100)
				Fin Para
				c=1
			SiNo
				Dimension vb(n)

					vb(i)=Aleatorio(-100,100)
				Fin Para
			FinSi
			Escribir "Se genero el Vector B"
			Esperar Tecla
		"c":
			Si n<>0 Entonces
				Dimension vc(n)
				Para i<-0 Hasta n-1 Con Paso 1 Hacer
					vc[i]=va[i]+vb[i]
				Fin Para
			SiNo
				Escribir "No genero el Vector A o Vector B"
				Esperar Tecla
			FinSi
		"d":
			Dimension vc(n)
			Para i<-0 Hasta n-1 Con Paso 1 Hacer
				vc[i]=vb[i]-va[i]
			Fin Para
		"e":
			Escribir "Que vector queire ver: A,B o C"
			Leer dat

				"A":

					Esperar Tecla

					Esperar Tecla

					Para i<-0 Hasta n-1 Con Paso 1 Hacer
						Escribir Sin Saltar "[",vc[i],"]"
					Fin Para

					Esperar Tecla
