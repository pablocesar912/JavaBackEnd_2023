Algoritmo ej3
	Definir v,n,m Como Entero
	Escribir "Ingrese el tamaño de la matriz cuadrada nxn"
	Leer n
	Dimension v[n,n]
	crear(v,n)
	ver(v,n)
FinAlgoritmo

SubProceso crear(v,n)
	Definir i,j Como Entero
	Para i<-0 Hasta n-1 Con Paso 1 Hacer
		Para j<-0 Hasta n-1 Con Paso 1 Hacer
			Si i=j Entonces
				v[i,j]=0
			SiNo
				v[i,j]=Aleatorio(0,9)
			FinSi
		Fin Para
	Fin Para
FinSubProceso

SubProceso ver(v,n)
	Definir i,j,suma Como Entero
	suma=0
	Escribir "La matriz generada"
	Para i<-0 Hasta n-1 Con Paso 1 Hacer
		Para j<-0 Hasta n-1 Con Paso 1 Hacer
			Escribir Sin Saltar v[i,j]," "
		Fin Para
		Escribir ""
	Fin Para
	Escribir ""
FinSubProceso
